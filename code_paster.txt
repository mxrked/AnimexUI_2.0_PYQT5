        def closeInfoPopup(popup):

            # Hiding the darken frame
            self.StartWindow_DarkenFrame.setMinimumHeight(0)
            self.StartWindow_DarkenFrame.setMaximumHeight(0)

            # Closing the info dialog
            popup.close()

                }
